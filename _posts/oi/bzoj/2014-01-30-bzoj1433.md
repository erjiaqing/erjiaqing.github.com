#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;
const int maxn=75;
vector<int> v[maxn];
typedef vector<int>::iterator it;
int bed[maxn],need[maxn],mat[maxn],use[maxn],n,totneed;
void reset()
{
    for (int i=1;i<maxn;i++)
        v[i].clear();
    memset(bed,0,sizeof(bed));
    memset(need,0,sizeof(need));
    memset(mat,0,sizeof(mat));
    memset(use,0,sizeof(use));
    n=totneed=0;
}
int find(int x)
{
    for (it i=v[x].begin();i!=v[x].end();i++)
    {
        if (!use[*i]&&bed[*i])
        {
            use[*i]=1;
            if (mat[*i]==0||find(mat[*i]))
            {
                mat[*i]=x;
                return 1;
            }
        }
    }
    return 0;
}
int xyl()
{
    int match=0;
    for (int i=1;i<=n;i++)
    {
        if (!need[i])
        {
            memset(use,0,sizeof(use));
            match+=find(i);
        }
    }
    return match;
}
void work()
{
    reset();
    int t;
    scanf("%d",&n);
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&need[i]);
        bed[i]=need[i];
    }
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&t);
        if (bed[i]&&!t)
            need[i]=0;
        if (!need[i])
            totneed++;
    }
    for (int i=1;i<=n;i++)
        for (int j=1;j<=n;j++)
        {
            scanf("%d",&t);
            if (t||(i==j))
                v[i].push_back(j);
        }
    if (xyl()==totneed)
        printf("^_^\n");
    else
        printf("T_T\n");
}
int main()
{
    int T;
    scanf("%d",&T);
    while (T--)
        work();
    return 0;
}
