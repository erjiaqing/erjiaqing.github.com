Q：为毛下面这个WA了,到n=46都是对的……

{% highlight cpp %}
#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
int n,m,ans;
const int mod=997;
int h[65],a[65],f[5005];
int g[65][65];
int gcd(int x,int y)
{
	if (y==0)
		return x;
	else
		return gcd(y,x%y);
}
int Pow(int n,int k)
{
	int r=1;
	for (;k;k>>=1,(n*=n)%=mod)
		if (k&1)
			(r*=n)%=mod;
	return r;
}
void dfs(int dep,int n,int last,int t,int l,int k)
{
	int tt,kk;
	if (n==0)
		(ans+=t*f[k])%=mod;
	else
	{
		tt=t;kk=k;
		for (int i=last;i<=n;i++)
		{
			a[dep]=i;t=tt;k=kk;
			t=t*f[i>>1]*h[i]%mod;
			for (int j=1;j<=dep-1;j++)(k+=g[a[j]][i])%=mod;
			if (i==a[dep-1])l++;else l=1;
			(t*=h[l])%=mod;
			dfs(dep+1,n-i,i,t,l,k);
		}
	}
}
int main()
{
	cin>>n;
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++)
			g[i][j]=gcd(i,j);
	f[0]=1;
	for (int i=1;i<=5000;i++)
		f[i]=(f[i-1]*2)%mod;
	for (int i=1;i<=n;i++)
		h[i]=Pow(i,mod-2);
	dfs(1,n,1,1,0,0);
	cout<<ans<<endl;
	return 0;
}
